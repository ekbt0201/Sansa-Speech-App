        <button id="prev-btn" class="nav-btn" onclick="changePage(-1)" disabled>⬅️ Back</button>
        
        <input type="color" id="color-picker" value="#ff0000"> <button class="tool-btn" onclick="clearCanvas()">🗑️ Clear</button>
        
        <button id="next-btn" class="nav-btn" onclick="changePage(1)">Next ➡️</button>

    <script>
        // Array holding all of your image files locally
        const pages = [
            '8301.png', '8302.png', '8303.png', '8304.png', 
            '8305.png', '8306.png', '8307.png', '8308.png', 
            '8309.png', '8310.png'
        ];
        
        let currentIndex = 0;
        
        // Setup Canvas and Drawing Tools
        const imgElement = document.getElementById('workbook-page');
        const canvas = document.getElementById('draw-layer');
        const ctx = canvas.getContext('2d');
        const colorPicker = document.getElementById('color-picker');
        
        let isDrawing = false;

        // Resize the canvas to perfectly match the image every time a page loads
        function resizeCanvas() {
            canvas.width = imgElement.clientWidth;
            canvas.height = imgElement.clientHeight;
            ctx.lineCap = 'round';
            ctx.lineJoin = 'round';
            ctx.lineWidth = 6; // Thickness of the crayon
        }

        // Wait for image to load before sizing canvas
        imgElement.onload = resizeCanvas;
        window.onresize = resizeCanvas;

        // Drawing Logic (Handles both Mouse and Touch)
        function startPosition(e) {
            isDrawing = true;
            draw(e);
        }

        function endPosition() {
            isDrawing = false;
            ctx.beginPath(); // Resets path so next line doesn't connect to the old one
        }

        function draw(e) {
            if (!isDrawing) return;
            
            // Get correct coordinates whether it's a mouse or a finger
            e.preventDefault(); 
            const rect = canvas.getBoundingClientRect();
            let clientX, clientY;

            if (e.touches && e.touches.length > 0) {
                clientX = e.touches[0].clientX;
                clientY = e.touches[0].clientY;
            } else {
                clientX = e.clientX;
                clientY = e.clientY;
            }

            const x = clientX - rect.left;
            const y = clientY - rect.top;

            ctx.strokeStyle = colorPicker.value; // Use selected color
            ctx.lineTo(x, y);
            ctx.stroke();
            ctx.beginPath();
            ctx.moveTo(x, y);
        }

        // Event Listeners for drawing
        canvas.addEventListener('mousedown', startPosition);
        canvas.addEventListener('mouseup', endPosition);
        canvas.addEventListener('mousemove', draw);
        canvas.addEventListener('mouseleave', endPosition);

        canvas.addEventListener('touchstart', startPosition, {passive: false});
        canvas.addEventListener('touchend', endPosition);
        canvas.addEventListener('touchmove', draw, {passive: false});

        // Navigation and Clearing
        function clearCanvas() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
        }

        function changePage(direction) {
            currentIndex += direction;
            imgElement.src = pages[currentIndex];
            
            // Clear drawing when turning the page
            clearCanvas(); 

            document.getElementById('prev-btn').disabled = currentIndex === 0;
            document.getElementById('next-btn').disabled = currentIndex === pages.length - 1;
            
            window.scrollTo(0, 0); 
        }
    </script>
